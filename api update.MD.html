เพิ่มข้อมูล

const params = {
    type: "add",
    car: "test",
    model: "test"
}
$.post("https://pasitdev.com/api/test", params, (data) => {
    // ค่าที่ API ตอบกลับมา
    {
        "status_code": 200,
        "message": "success"
    }
});

เข้าสู่ระบบ (Login)

const params = {
    type: "login",
    username: "first",
    password: "1234"
}
$.post("https://pasitdev.com/api/test", params, (data) => {
    // ค่าที่ API ตอบกลับมา
    {
        "status_code": 200,
        "message": "success",
        "user_id": 1,
        "username": "first",
        "full_name": "Itchaya"
    }
});
